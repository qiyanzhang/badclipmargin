   208	class CustomCLIP(nn.Module):
   209	    def __init__(self, cfg, classnames, clip_model):
   210	        super().__init__()
   211	        self.prompt_learner = PromptLearner(cfg, classnames, clip_model)
   212	        self.tokenized_prompts = self.prompt_learner.tokenized_prompts
   213	        self.image_encoder = clip_model.visual
   214	        self.text_encoder = TextEncoder(clip_model)
   215	        self.logit_scale = clip_model.logit_scale
   216	        self.dtype = clip_model.dtype
   217	
   218	        self.trigger = Trigger(cfg, self.dtype)
   219	        self.trigger.set_target_name(self.prompt_learner.classnames[int(self.trigger.target)])
   220	
   221	    def forward(self, image, label=None):
   222	        tokenized_prompts = self.tokenized_prompts
   223	        logit_scale = self.logit_scale.exp()
   224	
   225	        image_features = self.image_encoder(image.type(self.dtype))
   226	        image_features = image_features / image_features.norm(dim=-1, keepdim=True)
   227	
   228	        prompts = self.prompt_learner(image_features)
   229	        
   230	        logits = []
   231	        for pts_i, imf_i in zip(prompts, image_features):
   232	            text_features = self.text_encoder(pts_i, tokenized_prompts)
   233	            text_features = text_features / text_features.norm(dim=-1, keepdim=True)
   234	            l_i = logit_scale * imf_i @ text_features.t()
   235	            logits.append(l_i)
   236	        logits = torch.stack(logits)
   237	        
   238	        if self.prompt_learner.training:
   239	            return F.cross_entropy(logits, label)
   240	        
   241	        return logits
